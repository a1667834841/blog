在windows下编译Mac可执行程序

go env -w CGO_ENABLED=0
go env -w GOOS=darwin
go env -w GOARCH=amd64
go build main.go

在windows下发编译linux可执行程序

go env -w CGO_ENABLED=0
go env -w GOOS=linux
go env -w GOARCH=amd64
go build -o study_xxqg ./

nohup ./study_xxqg > ./study_xxqg.log 2>&1 & echo $!>pid.pid

恢复windows默认值
go env -w CGO_ENABLED=1
go env -w GOOS=windows
