```
<?xml version="1.0" encoding="UTF-8"?>
<!-- 日志级别从低到高分为TRACE < DEBUG < INFO < WARN < ERROR < FATAL，如果设置为WARN，则低于WARN的信息都不会输出 -->
<!-- scan:当此属性设置为true时，配置文件如果发生改变，将会被重新加载，默认值为true -->
<!-- scanPeriod:设置监测配置文件是否有修改的时间间隔，如果没有给出时间单位，默认单位是毫秒。当scan为true时，此属性生效。默认的时间间隔为1分钟。 -->
<!-- debug:当此属性设置为true时，将打印出logback内部日志信息，实时查看logback运行状态。默认值为false。 -->
<configuration>

    <include resource="org/springframework/boot/logging/logback/base.xml"/>

    <!--日志上下文名称-->
    <contextName>logback</contextName>
    <!--日志路径-->
    <property name="log.path" value="log/service-gift"/>
    <property name="format" value="%d{yyyy-MM-dd HH:mm:ss.SSS} %5p --- [%15.15t] %-40.40logger{39}[%line] : %m%n"/>


    <appender name="SQL_FILE" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <!--        <file>${LOG_HOME}/sql.log</file>-->
        <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
            <!--日志文件输出的文件名-->
            <FileNamePattern>${log.path}/sql/%d-%i.log</FileNamePattern>
            <!--日志文件保留天数-->
            <MaxHistory>30</MaxHistory>
            <!--日志文件最大的大小-->
            <MaxFileSize>300MB</MaxFileSize>
        </rollingPolicy>
        <encoder class="ch.qos.logback.classic.encoder.PatternLayoutEncoder">
            <!-- 格式化输出: %d: 日期; %-5level: 级别从左显示5个字符宽度; %thread: 线程名; %logger: 类名; %M: 方法名; %line: 行号; %msg: 日志消息; %n: 换行符 -->
            <pattern>[%d{yyyy-MM-dd HH:mm:ss.SSS}] [%-5level] [%thread] [%logger{50}] [%M] [%line] - %msg%n</pattern>
            <charset>UTF-8</charset>
        </encoder>

    </appender>

    <!-- 每天产生一个日志文件，最多 200 个，自动回滚 -->
    <appender name="uatFile" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <fileNamePattern>${log.path}/uat/shangye-service-gift/%d.log</fileNamePattern>
            <!--日志文件保留天数-->
            <maxHistory>200</maxHistory>
        </rollingPolicy>
        <encoder>
            <pattern>${format}</pattern>
            <charset>UTF-8</charset>
        </encoder>
    </appender>

    <appender name="prodFile" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <fileNamePattern>${log.path}/prod/shangye-service-gift/%d.log</fileNamePattern>
            <!--日志文件保留天数-->
            <maxHistory>200</maxHistory>
        </rollingPolicy>
        <encoder>
            <pattern>${format}</pattern>
            <charset>UTF-8</charset>
        </encoder>
    </appender>

    <appender name="consoleLog" class="ch.qos.logback.core.ConsoleAppender">
        <layout class="ch.qos.logback.classic.PatternLayout">
            <pattern>[%d{yyyy-MM-dd HH:mm:ss.SSS}] [%-5level] [%thread] [%logger{50}] [%M] [%line] - %msg%n</pattern>
        </layout>
    </appender>



    <appender name="testFile" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <fileNamePattern>${log.path}/test/shangye-service-gift/%d.log</fileNamePattern>
            <!--日志文件保留天数-->
            <maxHistory>30</maxHistory>
        </rollingPolicy>
        <encoder>
            <pattern>${format}</pattern>
            <charset>UTF-8</charset>
        </encoder>
    </appender>



    <!--test 环境日志配置-->
    <springProfile name="test">
        <logger name="org.springframework.web" level="INFO"/>
        <logger name="org.springboot.sample" level="INFO"/>
        <logger name="com.daoism" level="INFO"/>
        <logger name="com.daoism.service.gift.ServiceGiftApplication" level="INFO"/>
        <logger name="org.springframework.boot.web.embedded.tomcat.TomcatWebServer" level="INFO"/>
        <logger name="io.swagger.models.parameters.AbstractSerializableParameter" level="INFO"/>
        <logger name="com.daoism.service.gift.dao" level="DEBUG" additivity="false">
            <appender-ref ref="SQL_FILE"/>
        </logger>
        <!-- 日志输出级别 -->
        <root level="INFO">
            <appender-ref ref="testFile"/>
        </root>
    </springProfile>

    <!-- 开发 环境 -->
    <springProfile name="dev">
        <logger name="org.springframework.web" level="DEBUG"/>
        <logger name="org.springboot.sample" level="DEBUG"/>
        <logger name="com.daoism" level="DEBUG"/>
        <logger name="io.swagger.models.parameters.AbstractSerializableParameter" level="ERROR"/>

    </springProfile>

    <!-- 测试、验收、生产 环境 -->
    <springProfile name="uat">
        <logger name="org.springframework.web" level="ERROR"/>
        <logger name="org.springboot.sample" level="ERROR"/>
        <logger name="com.daoism" level="ERROR"/>
        <logger name="com.daoism.service.gift.ServiceGiftApplication" level="INFO"/>
        <logger name="org.springframework.boot.web.embedded.tomcat.TomcatWebServer" level="INFO"/>
        <logger name="io.swagger.models.parameters.AbstractSerializableParameter" level="ERROR"/>
        <logger name="com.daoism.service.gift.dao" level="DEBUG" additivity="false">
            <appender-ref ref="SQL_FILE"/>
        </logger>

        <!-- 日志输出级别 -->
        <root level="ERROR">
            <appender-ref ref="uatFile"/>
        </root>
    </springProfile>

    <springProfile name="prod">
        <logger name="org.springframework.web" level="ERROR"/>
        <logger name="org.springboot.sample" level="ERROR"/>
        <logger name="com.daoism" level="ERROR"/>
        <logger name="com.daoism.service.gift.ServiceGiftApplication" level="INFO"/>
        <logger name="org.springframework.boot.web.embedded.tomcat.TomcatWebServer" level="INFO"/>
        <logger name="io.swagger.models.parameters.AbstractSerializableParameter" level="ERROR"/>
        <logger name="com.daoism.service.gift.dao" level="DEBUG" additivity="false">
            <appender-ref ref="SQL_FILE"/>
        </logger>

        <!-- 日志输出级别 -->
        <root level="ERROR">
            <appender-ref ref="prodFile"/>
        </root>
    </springProfile>




</configuration>
```
